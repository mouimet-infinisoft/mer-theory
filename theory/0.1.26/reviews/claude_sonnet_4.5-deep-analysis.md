# Design of Experiments (DoE) approach
def parameter_sweep():
    """
    Systematic exploration of parameter space.
    Using Latin Hypercube Sampling for efficiency.
    """
    from scipy.stats import qmc
    
    # Parameters to vary
    param_ranges = {
        'expansion_coeff': [1.0, 2.0],      # Test φ=1.618 vs others
        'regulation_coeff': [-1.0, 0.0],    # Test ψ=-0.618 vs others
        'coupling_strength': [0.1, 10.0],
        'nonlinearity': [1.0, 3.0]
    }
    
    # Generate 1000 parameter combinations (LHS)
    sampler = qmc.LatinHypercube(d=4)
    samples = sampler.random(n=1000)
    
    # Scale to parameter ranges
    params = {}
    for i, (name, (low, high)) in enumerate(param_ranges.items()):
        params[name] = low + samples[:, i] * (high - low)
    
    results = []
    for i in range(1000):
        system = PhiPsiSystem3D(
            phi=params['expansion_coeff'][i],
            psi=params['regulation_coeff'][i],
            coupling=params['coupling_strength'][i],
            nl_order=params['nonlinearity'][i]
        )
        
        # Run simulation
        trajectory = system.evolve(t_max=1000, dt=0.01)
        
        # Compute metrics
        metrics = {
            'lyapunov_max': compute_lyapunov(trajectory),
            'fractal_dim': fractal_dimension(trajectory),
            'attractor_volume': attractor_metrics(trajectory)['volume'],
            'symmetry_index': measure_symmetry(trajectory),
            'lemniscate_score': match_lemniscate_topology(trajectory)
        }
        
        results.append({
            'params': {k: v[i] for k, v in params.items()},
            'metrics': metrics
        })
    
    return results

def analyze_phi_significance(results):
    """
    Statistical test: Is φ/ψ special?
    """
    import pandas as pd
    from scipy import stats
    
    df = pd.DataFrame(results)
    
    # Identify golden ratio cases (within 1% tolerance)
    phi_cases = df[
        (np.abs(df['params.expansion_coeff'] - 1.618) < 0.016) &
        (np.abs(df['params.regulation_coeff'] + 0.618) < 0.006)
    ]
    
    other_cases = df[~df.index.isin(phi_cases.index)]
    
    # Statistical comparison
    tests = {}
    for metric in ['lyapunov_max', 'fractal_dim', 'lemniscate_score']:
        t_stat, p_value = stats.ttest_ind(
            phi_cases[f'metrics.{metric}'],
            other_cases[f'metrics.{metric}']
        )
        tests[metric] = {
            't_statistic': t_stat,
            'p_value': p_value,
            'significant': p_value < 0.05,
            'effect_size': (
                phi_cases[f'metrics.{metric}'].mean() - 
                other_cases[f'metrics.{metric}'].mean()
            ) / phi_cases[f'metrics.{metric}'].std()
        }
    
    return tests

class TestKnownAttractors:
    """
    Validation suite: Reproduce published results.
    """
    
    def test_lorenz_attractor(self):
        """
        Can we recover Lorenz attractor with specific φ/ψ parameters?
        Expected: σ=10, ρ=28, β=8/3
        MER mapping: φ→σ, ψ→-β, ...
        """
        # Standard Lorenz
        from scipy.integrate import odeint
        
        def lorenz(state, t, sigma=10, rho=28, beta=8/3):
            x, y, z = state
            return [
                sigma * (y - x),
                x * (rho - z) - y,
                x * y - beta * z
            ]

        lorenz_traj = odeint(lorenz, [1, 1, 1], t)
        
        # MER equivalent (if it exists)
        mer_system = PhiPsiSystem3D.from_standard_system(
            system_type='lorenz',
            params={'sigma': 10, 'rho': 28, 'beta': 8/3}
        )
        mer_traj = mer_system.evolve(t_max=50, dt=0.005)
        
        # Compare

        information is lost proportional to (system_scale/cell_size).
        """

        # we lose information about L_S/L_O sub-cells.

"""
Vertical Slice 1.1: One-dimensional φ/ψ dynamics

Question: What is the simplest φ/ψ iterative system?
Answer: x_{n+1} = φ·f(x_n) + ψ·g(x_n)

This module:
- Defines the 1D system explicitly
- Proves boundedness
- Visualizes behavior
- Validates against known maps (logistic map, tent map)
"""

import numpy as np
import matplotlib.pyplot as plt
from typing import Callable, Tuple

class PhiPsi1D:
    """Minimal φ/ψ dynamical system in 1D."""
    
    PHI = (1 + np.sqrt(5)) / 2  # 1.618...
    PSI = -1 / PHI              # -0.618...
    
    def __init__(self, 
                 f_expand: Callable[[float], float],
                 f_regulate: Callable[[float], float]):
        """
        Args:
            f_expand: Expansion function (φ-weighted)
            f_regulate: Regulation function (ψ-weighted)
        """
        self.f_expand = f_expand
        self.f_regulate = f_regulate
    
    def iterate(self, x0: float, n_steps: int) -> np.ndarray:
        """Iterate the system n_steps times."""
        trajectory = np.zeros(n_steps)
        trajectory[0] = x0
        
        for i in range(1, n_steps):
            trajectory[i] = (
                self.PHI * self.f_expand(trajectory[i-1]) +
                self.PSI * self.f_regulate(trajectory[i-1])
            )
        
        return trajectory
    
    def find_fixed_points(self) -> list:
        """Find fixed points: x* = φ·f(x*) + ψ·g(x*)"""
        # Numerical root finding
        from scipy.optimize import fsolve
        
        def equation(x):
            return x - (self.PHI * self.f_expand(x) + 
                       self.PSI * self.f_regulate(x))
        
        # Try multiple initial guesses
        fixed_points = []
        for x0 in np.linspace(-10, 10, 50):
            fp = fsolve(equation, x0)[0]
            if abs(equation(fp)) < 1e-6:
                # Check if already found
                if not any(abs(fp - existing) < 1e-4 
                          for existing in fixed_points):
                    fixed_points.append(fp)
        
        return sorted(fixed_points)
    
    def stability_analysis(self, fixed_point: float) -> dict:
        """Analyze stability of a fixed point."""
        # Compute derivative numerically
        h = 1e-8
        f_prime = (
            (self.PHI * self.f_expand(fixed_point + h) +
             self.PSI * self.f_regulate(fixed_point + h)) -
            (self.PHI * self.f_expand(fixed_point - h) +
             self.PSI * self.f_regulate(fixed_point - h))
        ) / (2 * h)
        
        return {
            'fixed_point': fixed_point,
            'derivative': f_prime,
            'stable': abs(f_prime) < 1,
            'type': 'stable' if abs(f_prime) < 1 else 'unstable'
        }

"""
Vertical Slice 1.2: Two-dimensional φ/ψ dynamics

Question: Can φ/ψ produce 2D attractors?
Approach: Extend 1D to coupled 2D system

System:
  x_{n+1} = φ·f(x_n, y_n) + ψ·g(x_n, y_n)

"""

import numpy as np
import matplotlib.pyplot as plt
from dataclasses import dataclass
from typing import Tuple, Callable

@dataclass
class PhiPsi2DConfig:
    """Configuration for 2D φ/ψ system."""
    f_expand: Callable[[float, float], float]
    g_regulate: Callable[[float, float], float]
    h_expand: Callable[[float, float], float]
    k_regulate: Callable[[float, float], float]

class PhiPsi2D:
    """Two-dimensional φ/ψ dynamics."""
    
    PHI = (1 + np.sqrt(5)) / 2
    PSI = -1 / PHI
    
    def __init__(self, config: PhiPsi2DConfig):
        self.config = config
    
    def iterate(self, 
                x0: float, y0: float, 
                n_steps: int) -> Tuple[np.ndarray, np.ndarray]:
        """Evolve 2D system."""
        x = np.zeros(n_steps)
        y = np.zeros(n_steps)
        x[0], y[0] = x0, y0
        
        for i in range(1, n_steps):
            x[i] = (self.PHI * self.config.f_expand(x[i-1], y[i-1]) +
                   self.PSI * self.config.g_regulate(x[i-1], y[i-1]))
            y[i] = (self.PHI * self.config.h_expand(x[i-1], y[i-1]) +
                   self.PSI * self.config.k_regulate(x[i-1], y[i-1]))
        
        return x, y
    
    @classmethod
    def henon_like(cls, a: float = 1.4, b: float = 0.3):
        """
        Create Hénon-like system using φ/ψ.
        
        Standard Hénon:
          x_{n+1} = 1 - a·x_n² + y_n
          y_{n+1} = b·x_n
        
        φ/ψ interpretation:
          Try to map parameters to φ/ψ structure
        """
        config = PhiPsi2DConfig(
            f_expand=lambda x, y: 1 + y - a*x**2,
            g_regulate=lambda x, y: 0,
            h_expand=lambda x, y: b*x,
            k_regulate=lambda x, y: 0
        )
        return cls(config)
    
    @classmethod
    def coupled_logistic(cls, r: float = 3.8, coupling: float = 0.1):
        """
        Coupled logistic maps with φ/ψ structure.
        """
        config = PhiPsi2DConfig(
            f_expand=lambda x, y: r*x*(1-x) + coupling*y,
            g_regulate=lambda x, y: 0,
            h_expand=lambda x, y: r*y*(1-y) + coupling*x,
            k_regulate=lambda x, y: 0
        )
        return cls(config)

PHI = (1 + np.sqrt(5)) / 2

def load_nist_data(element: str) -> np.ndarray:

    # NIST provides CSV exports
    # https://physics.nist.gov/PhysRefData/ASD/levels_form.html
    url = f"https://physics.nist.gov/cgi-bin/ASD/energy1.pl?...{element}"
    data = pd.read_csv(url)
    energies = data['Level (eV)'].values
    return energies[energies > 0]  # Positive energies only

def test_phi_spacing(energies: np.ndarray) -> dict:
    """
    Test if energies follow E_n = E_0 * φ^n pattern.
    
    Returns:
        - hypothesis: "φ-spaced" or "not φ-spaced"
        - p_value: statistical significance
        - deviation: mean % error from φ^n
    """
    # Sort energies
    E = np.sort(energies)
    E0 = E[0]  # Ground state
    
    # Compute ratios to ground state
    ratios = E / E0
    
    # Find best-fit n values (if φ^n pattern exists)
    # E_n / E_0 = φ^n  →  n = log(E_n/E_0) / log(φ)
    n_predicted = np.log(ratios) / np.log(PHI)
    n_integer = np.round(n_predicted)
    
    # Expected energies if φ^n holds
    E_expected = E0 * PHI ** n_integer
    
    # Compute deviation
    deviation_pct = np.abs((E - E_expected) / E) * 100
    mean_deviation = np.mean(deviation_pct)
    
    # Statistical test: Are deviations random or systematic?
    # If random → φ pattern holds
    # If systematic → φ pattern is coincidence
    
    # Chi-squared test
    chi2 = np.sum((E - E_expected)**2 / E_expected)
    p_value = stats.chi2.sf(chi2, df=len(E)-1)
    
    # Also test against random power law (e^n, 2^n, π^n)

        'e': np.e,
        '2': 2.0,
        'π': np.pi,
        '√2': np.sqrt(2),
        '√3': np.sqrt(3)
    }

def analyze_lorenz_for_phi():
    """
    Analyze Lorenz attractor WITHOUT assuming φ.
    
    What to measure:
    - Ratio of lobe sizes
    - Frequency ratios
    - Return map slopes
    - Fractal dimension
    """
    from scipy.integrate import odeint
    
    def lorenz(state, t, sigma=10, rho=28, beta=8/3):
        x, y, z = state
        return [
            sigma * (y - x),
            x * (rho - z) - y,
            x * y - beta * z
        ]
    
    # Solve
    t = np.linspace(0, 100, 100000)
    xyz = odeint(lorenz, [1, 1, 1], t)
    
    # Measure lobe ratio (left vs right)
    left_lobe = xyz[xyz[:, 0] < 0]
    right_lobe = xyz[xyz[:, 0] > 0]
    
    ratio_volumes = len(right_lobe) / len(left_lobe)
    
    # Measure crossing frequencies
    crossings = np.where(np.diff(np.sign(xyz[:, 0])))[0]
    times_between = np.diff(t[crossings])
    freq_ratio = np.mean(times_between[1:] / times_between[:-1])
    
    # Does natural Lorenz exhibit φ?
    results = {
        'lobe_ratio': ratio_volumes,
        'frequency_ratio': freq_ratio,
        'phi_in_lobes': abs(ratio_volumes - PHI) < 0.1,
        'phi_in_frequency': abs(freq_ratio - PHI) < 0.1
    }
    
    return results
